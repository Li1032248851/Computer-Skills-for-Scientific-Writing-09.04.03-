	\begin{tikzpicture}[samples=100, scale=1.5, >=stealth]
		% 1. 绘制坐标轴
		\draw[->, gray] (-2.2, 0) -- (2.5, 0) node[right] {$x$};
		\draw[->, gray] (0, -2.2) -- (0, 4.5) node[above] {$y$};
		
		% 2. 标注原点
		\node[below left, gray] at (0,0) {$O$};
		\filldraw[gray] (0,0) circle (1pt);
		
		% 3. 绘制 y = e^x (蓝色)
		\draw[blue, thick, domain=-2:1.5] plot (\x, {exp(\x)}) node[right] {$y = e^x$};
		
		% 4. 绘制 y = ln(x) (黑色) - 修复处：改为 ln(\x)
		\draw[black, thick, domain=0.1:2.2] plot (\x, {ln(\x)}) node[right] {$y = \ln(x)$};
		
		% 5. 绘制关键点参考线
		\draw[ultra thin, gray] (-0.1, 1) -- (0.1, 1) node[left] {$y = 1$};
		\draw[ultra thin, gray] (1, -0.1) -- (1, 0.1) node[below] {$x = 1$};
		
	\end{tikzpicture}
