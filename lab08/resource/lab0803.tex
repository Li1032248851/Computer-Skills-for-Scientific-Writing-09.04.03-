\documentclass[border=1cm]{standalone}
\usepackage{tikz}

\begin{document}
	
	\begin{tikzpicture}[scale=5]
		% 修复版递归宏：使用 \numexpr 进行数学计算，确保 \ifnum 识别
		\newcommand{\carpet}[4]{
			\ifnum#1>0
			\foreach \i in {0,1,2} {
				\foreach \j in {0,1,2} {
					% 判断是否为中心正方形 (1,1)
					% Определите, является ли центральный квадрат (1,1)
					\ifnum \numexpr \i*3+\j \relax = 4
					% 中心块：不进行任何操作（跳过）
					% Центральный блок: Операция не выполнялась (пропустить)
					\else
					% 递归调用：计算下一层的参数
					% Рекурсивный вызов: вычислить параметры следующего уровня
					\carpet{\numexpr#1-1\relax}{#2+\i*#4/3}{#3+\j*#4/3}{#4/3}
					\fi
				}
			}
			\else
			% 底层：绘制填充正方形
			 %Нижний слой: Нарисуйте закрашенный квадрат.
			\fill[blue!70!black] (#2,#3) rectangle ++(#4,#4);
			\fi
		}
		
		% 运行：3层迭代，起点(0,0)，边长1
		% Выполнение: 3-уровневая итерация, начальная точка (0,0), длина стороны 1
		\carpet{3}{0}{0}{1}
	\end{tikzpicture}
	
\end{document}
